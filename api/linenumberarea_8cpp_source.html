00034     QColor bg = palette().color(QPalette::Background);

00038         bg = bg.lighter(120);
00039     }
00040     painter.fillRect(event->rect(), bg);
00041 
00042     QTextBlock block = ionText->getEditorInfo().firstVisibleBlock();
00043     int blockNumber = block.blockNumber();
00044     double top = (int) ionText->getEditorInfo().blockBoundingGeometry(block).translated(ionText->getEditorInfo().contentOffset()).top();
00045     double bottom = top + (int) ionText->getEditorInfo().blockBoundingRect(block).height();
00046 
00047     QFont font("Monaco");
00048     font.setStyleHint(QFont::Courier, QFont::PreferAntialias);
00049     font.setPointSize(12);
00050     painter.setFont(font);
00051     int fontHeight = painter.fontMetrics().height();
00052 
00053     bool resetColor = true;
00054     while (block.isValid() && top <= event->rect().bottom()) {
00055         double blockHeight = ionText->getEditorInfo().blockBoundingRect(block).height();
00056 
00057         if (block.isVisible() && bottom >= event->rect().top()) {
00058             if (blockNumber == currentLine) {
00059                 painter.setPen(QColor(0x70, 0x70, 0x70));
00060                 resetColor = true;
00061             } else if (resetColor) {
00062                 painter.setPen(QColor(0xb0, 0xb0, 0xb0));
00063                 resetColor = false;
00064             }
00065 
00066             painter.drawText(
00067                 QRectF(
00068                     3,
00069                     top + (blockHeight / block.lineCount() - fontHeight) / 2.0f - 1,
00070                     this->width()-5,
00071                     blockHeight
00072                 ),
00073                 Qt::AlignRight,
00074                 QString::number(blockNumber + 1)
00075             );
00076         }
00077 
00078         block = block.next();
00079         top = bottom;
00080         bottom = top + blockHeight;
00081         ++blockNumber;
00082     }
