00033 void LayoutZonesManager::addZone(ZoneDefinition &zone)
00034 {
00035    ZoneNodeBranch *parentNode = root->getZone(zone.parentPath)->getZoneAsBranch();
00036    Q_ASSERT(parentNode);
00037    ZoneNodeLeaf *leaf = new ZoneNodeLeaf(parentNode, zone);
00038    int pos;
00039    signed int index = -1;
00040 
00041    if ( zone.after.length() && ((pos = parentNode->indexOf(zone.after)) >= 0) ) {
00042        index = 1 + pos;
00043    } else if ( zone.before.length() && ((pos = parentNode->indexOf(zone.before)) >= 0) ) {
00044        index = pos;
00045    }
00046    parentNode->addSubZone(leaf, index);
00047 
00048    if (zone.hideIfEmpty) {
00049        leaf->getWidget()->hide();
00050    } else {
00051        leaf->show();
